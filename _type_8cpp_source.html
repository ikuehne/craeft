95 /* Visitor for converting Craeft types to LLVM types. */
96 struct ToLlvmVisitor: public boost::static_visitor<llvm::Type *> {
97  ToLlvmVisitor(llvm::Module &mod): ctx(mod.getContext()),
98  mod(mod) {}
99 
100  template<typename T>
101  llvm::Type *operator()(const T &t) const {
102  return t.to_llvm(ctx);
103  }
104 
105  llvm::Type *operator()(const Function<Type> &func) const {
106  std::vector<llvm::Type *> arg_types;
107 
108  for (const auto &t: func.get_args()) {
109  arg_types.push_back(to_llvm_type(*t, mod));
110  }
111 
112  auto *ret = to_llvm_type(*func.get_rettype(), mod);
113 
114  return llvm::FunctionType::get(ret, arg_types, false);
115  }
116 
117  llvm::Type *operator()(const Pointer<Type> &ptr) const {
118  return llvm::PointerType::getUnqual(to_llvm_type(*ptr.get_pointed(),
119  mod));
120  }
121 
122  llvm::Type *operator()(const Struct<Type> &str) const {
123  auto *result = mod.getTypeByName(str.get_name());
124  if (result) {
125  return result;
126  }
127 
128  std::vector<llvm::Type *> arg_types;
129 
130  for (const auto &field: str.get_fields()) {
131  arg_types.push_back(to_llvm_type(*field.second, mod));
132  }
133 
134  result = llvm::StructType::create(arg_types, str.get_name());
135 
136  return result;
137  }
138 
139 private:
140  llvm::LLVMContext &ctx;
141  llvm::Module &mod;
142 };
