212 class ToplevelVisitor {
213 public:
214  virtual ~ToplevelVisitor(void) {}

216  Result visit(const Toplevel &t) {
217  switch (t.kind()) {
218 #define HANDLE(X) case Toplevel::ToplevelKind::X:\
219  return operator()(llvm::cast<X>(t));
220  HANDLE(TypeDeclaration);
221  HANDLE(StructDeclaration);
222  HANDLE(TemplateStructDeclaration);
223  HANDLE(FunctionDeclaration);
224  HANDLE(FunctionDefinition);
225  HANDLE(TemplateFunctionDefinition);
226  }
227 #undef HANDLE
228  }
229 
230 private:

233  virtual Result operator()(const TemplateStructDeclaration &) = 0;
