147 Value ExpressionCodegen::operator()(
148  const std::unique_ptr<AST::Binop> &binop) {
149  auto lhs = codegen(binop->lhs);
150  auto rhs = codegen(binop->rhs);
151 
152  auto pos = binop->pos;
153 
154  if (binop->op == "<<") {
155  return translator.left_shift(lhs, rhs, pos);
156  } else if (binop->op == ">>") {
157  return translator.right_shift(lhs, rhs, pos);
158  } else if (binop->op == "&") {
159  return translator.bit_and(lhs, rhs, pos);
160  } else if (binop->op == "|") {
161  return translator.bit_or(lhs, rhs, pos);
162  } else if (binop->op == "^") {
163  return translator.bit_xor(lhs, rhs, pos);
164  } else if (binop->op == "+") {
165  return translator.add(lhs, rhs, pos);
166  } else if (binop->op == "-") {
167  return translator.sub(lhs, rhs, pos);
168  } else if (binop->op == "*") {
169  return translator.mul(lhs, rhs, pos);
170  } else if (binop->op == "/") {
171  return translator.div(lhs, rhs, pos);
172  } else if (binop->op == "==") {
173  return translator.equal(lhs, rhs, pos);
174  } else if (binop->op == "!=") {
175  return translator.nequal(lhs, rhs, pos);
176  } else if (binop->op == "<") {
177  return translator.less(lhs, rhs, pos);
178  } else if (binop->op == "<=") {
179  return translator.lesseq(lhs, rhs, pos);
180  } else if (binop->op == ">") {
181  return translator.greater(lhs, rhs, pos);
182  } else if (binop->op == ">=") {
183  return translator.greatereq(lhs, rhs, pos);
184  } else if (binop->op == "&&") {
185  return translator.bool_and(lhs, rhs, pos);
186  } else if (binop->op == "||") {
187  return translator.bool_or(lhs, rhs, pos);
188  } else {
189  throw Error("internal error", "unrecognized operator \"" + binop->op
190  + "\"",
191  fname, pos);
192 
193  }
194 }

273 void StatementCodegen::operator()(
274  const std::unique_ptr<AST::IfStatement> &if_stmt) {
275  /* Generate code for the condition. */
276  auto cond = expr_codegen.codegen(if_stmt->condition);
277 
278  // TODO: add appropriate abstraction to translator.
279  auto &builder = translator.get_builder();
280  auto &context = translator.get_ctx();
281  auto &env = translator.get_env();
282 
283  llvm::Function *parent = builder.GetInsertBlock()->getParent();
284 
285  auto *then_bb = llvm::BasicBlock::Create(context, "then", parent);
286  auto *else_bb = llvm::BasicBlock::Create(context, "else");
287  auto *merge_bb = llvm::BasicBlock::Create(context, "merge");
288 
289  assert(then_bb);
290  assert(else_bb);
291  assert(merge_bb);
292 
293  builder.CreateCondBr(cond.to_llvm(), then_bb, else_bb);
294 
295  // Generate "then" code.
296  builder.SetInsertPoint(then_bb);
297  // In a scope...
298  env.push();
299  for (const auto &arg: if_stmt->if_block) {
300  codegen(arg);
301  }
302  // which we forget after exiting.
303  env.pop();
304 
305  if (builder.GetInsertBlock()->getTerminator() == nullptr) {
306  builder.CreateBr(merge_bb);
307  }
308 
309  then_bb = builder.GetInsertBlock();
310 
311  // Generate "else" code.
312  parent->getBasicBlockList().push_back(else_bb);
313  builder.SetInsertPoint(else_bb);
314  env.push();
315  for (const auto &arg: if_stmt->else_block) {
316  codegen(arg);
317  }
318  env.pop();
319  if (builder.GetInsertBlock()->getTerminator() == nullptr) {
320  builder.CreateBr(merge_bb);
321  }
322 
323  // Start again at "merge".
324  parent->getBasicBlockList().push_back(merge_bb);
325  builder.SetInsertPoint(merge_bb);
326 }

364 Function ModuleCodegenImpl::type_of_ast_decl(
365  const AST::FunctionDeclaration &fd) {
366  std::vector<std::shared_ptr<Type> > arg_types;
367  TypeCodegen tg(translator, fname);
368 
369  for (const auto &decl: fd.args) {
370  arg_types.push_back(std::make_shared<Type>
371  (tg.codegen(decl->type)));
372  }
373 
374  auto ret_type = std::make_shared<Type>
375  (tg.codegen(fd.ret_type));
376 
377  return Function(ret_type, arg_types);
378 }
