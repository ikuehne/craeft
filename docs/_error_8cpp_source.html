52  if (files_read.count(f)) {
53  return *files_read[f];
54  } else {
55  auto v = std::make_unique<std::vector<std::string>>();
56  std::ifstream file(f);
57  char buf[81];
58  while (!file.eof()) {
59  unsigned i;
60  char c;
61  for (i = 0; i < 80; ++i) {
62  if (file.eof()) break;
63  c = file.get();
64  if (c == '\n') {
65  break;
66  }
67  buf[i] = c;
68  }
69  if (c != '\n') {
70  while (!file.eof() && file.get() != '\n');
71  }
72  buf[i] = '\0';
73  v->push_back(std::string(buf));
74  }
75  files_read[f] = std::move(v);
