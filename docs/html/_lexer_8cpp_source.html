59 boost::variant<double, uint64_t> Lexer::lex_number(void) {
60  uint64_t num = 0;
61 
62  /* Parse the initial series of digits. */
63  while (isdigit(c)) {
64  num *= 10;
65  num += digit(c);
66  get();
67  }
68 
69  /* If a decimal, */
70  if (c == '.') {
71  /* Parse the decimal bit. */
72  double result = (double)num;
73  double decimal_places = 0.1;
74  for (get(); isdigit(c); get()) {
75  result += decimal_places * digit(c);
76  decimal_places *= 0.1;
77  }
78 
79 
80  /* Parse the exponent if present. */
81  if (c == 'e' || c == 'E') {
82  get();
83 
84  bool neg = false;
85 
86  if (c == '-') {
87  get();
88  neg = true;
89  }
90 
91  uint64_t exp = 0;
92  for (; isdigit(c); get()) {
93  exp *= 10;
94  exp += digit(c);
95  }
96 
97  return result * pow(10.0, (neg? -1: 1) * exp);
98  }
99 
100  return result;
101  } else if (c == 'e' || c == 'E') {
102  get();
103 
104  bool neg = false;
105  if (c == '-') {
106  get();
107  neg = true;
108  }
109 
110  double result = (double)num;
111 
112  double exp = 0;
113  for (; isdigit(c); get()) {
114  exp *= 10.0;
115  exp += (double)digit(c);
116  }

124 void Lexer::shift(void) {
125  while (std::isspace(c)) {
126  get();
127  }
128 
129  if (c == std::char_traits<char>::eof()) {
130  eof = true;
131  return;
132  }
133 
134  /* Type name. */
135  if (isupper(c)) {
136  std::string tname;
137 
138  while (isalpha(c)) {
139  tname.push_back(c);
140  get();
141  }
142 
143  tok = Tok::TypeName(tname);
144  /* Identifiers and identifier-like keywords. */
145  } else if (islower(c)) {
146  std::string ident;
147 
148  while (isalpha(c)) {
149  ident.push_back(c);
150  get();
151  }
152 
153  /* Keywords that otherwise look like identifiers. */
154  if (ident == "fn") tok = Tok::Fn();
155  else if (ident == "struct") tok = Tok::Struct();
156  else if (ident == "return") tok = Tok::Return();
157  else if (ident == "if") tok = Tok::If();
158  else if (ident == "else") tok = Tok::Else();
159  else if (ident == "while") tok = Tok::While();
160  /* If none of those, an identifier. */
161  else tok = Tok::Identifier(ident);
162  /* Numeric literal. */
163  } else if (isdigit(c)) {
164  auto result = lex_number();
165  if (result.which() == 0) {
166  tok = Tok::FloatLiteral(boost::get<double>(result));
167  } else {
168  tok = Tok::UIntLiteral(boost::get<uint64_t>(result));
169  }
170  /* Operators. This is easily extensible to user-defined operators. */
171  } else if (is_opchar(c)) {
172  std::string result;
173  result.push_back(c);
174 
175  for (get(); is_opchar(c); get()) {
176  result.push_back(c);
177  }
178 
179  tok = Tok::Operator(result);
180  /* Some random syntax. */
181  } else if (c == '(') {
182  tok = Tok::OpenParen();
183  get();
184  } else if (c == ')') {
185  tok = Tok::CloseParen();
186  get();
187  } else if (c == '{') {
188  tok = Tok::OpenBrace();
189  get();
190  } else if (c == '}') {
191  tok = Tok::CloseBrace();
192  get();
193  } else throw "Character not recognized";
194 }
