52 AST::Expression Parser::parse_variable(void) {
53  auto tok = boost::get<Tok::Identifier>(lexer.get_tok());
54 
55  std::string id = tok.name;
56 
57  // Shift the name.
58  lexer.shift();
59 
60  /* Case not function call. */
61  if (!is_type<Tok::OpenParen>(lexer.get_tok())) {
62  return AST::Variable(id, lexer.get_pos());
63  }
64 
65  // Shift the opening paren.
66  lexer.shift();
67 
68  // Accumulate vector of args.
69  std::vector<AST::Expression> args;
70 
71  if (!is_type<Tok::CloseParen>(lexer.get_tok())) {
72  while (true) {
73  AST::Expression arg(parse_expression());
74  args.push_back(std::move(arg));
75 
76  /* If close paren, we've reached the end of the list. */
77  if (is_type<Tok::CloseParen>(lexer.get_tok())) {
78  break;
79  }
80 
81  // Expect comma before next arg.
82  if (!is_type<Tok::Comma>(lexer.get_tok())) {
83  throw ("Expected comma or close paren in "
84  "function argument list.");
85  }
86 
87  // Shift the comma.
88  lexer.shift();
89  }
90  }
91 
92  // Shift the close paren.
93  lexer.shift();
94 
95  return std::make_unique<AST::FunctionCall>(
96  std::move(id), std::move(args), lexer.get_pos());
97 }

99 AST::Expression Parser::parse_binop(int prec, AST::Expression lhs) {
100  auto start = lexer.get_pos();
101 
102  while (true) {
103  int old_prec = get_token_precedence();
104 
105  if (old_prec < prec) return lhs;
106 
107  // Thing in expression was not an operator.
108  if (!is_type<Tok::Operator>(lexer.get_tok())) {
109  assert(lexer.get_tok().which() != 5);
110  throw "Expected operator in arithmetic expression";
111  }
112 
113  auto op = boost::get<Tok::Operator>(lexer.get_tok());
114 
115  lexer.shift();
116 
117  auto rhs = parse_primary();
118 
119  int new_prec = get_token_precedence();
120 
121  if (old_prec < new_prec) {
122  rhs = parse_binop(old_prec + 1, std::move(rhs));
123  }
124 
125  lhs = std::make_unique<AST::Binop>(
126  op.op, std::move(lhs), std::move(rhs), start);
127  }
128 }

130 std::unique_ptr<AST::Cast> Parser::parse_cast(void) {
131 
132  auto start = lexer.get_pos();
133  auto tname = boost::get<Tok::TypeName>(lexer.get_tok());
134 
135  // Shift the typename.
136  lexer.shift();
137 
138  // No closing parenthesis.
139  if (!is_type<Tok::CloseParen>(lexer.get_tok())) {
140  throw "Expected close paren after type in cast.";
141  }
142 
143  // Shift the closing paren.
144  lexer.shift();
145 
146  auto expr = parse_expression();
147 
148  auto t = std::make_unique<Type>(std::move(tname.name));
149 
150  return std::make_unique<AST::Cast>(std::move(t), std::move(expr), start);
151 }

153 AST::Expression Parser::parse_parens(void) {
154  auto start = lexer.get_pos();
155 
156  // Shift the opening paren.
157  lexer.shift();
158 
159  // Might be a cast.
160  if (is_type<Tok::TypeName>(lexer.get_tok())) {
161  auto cast = parse_cast();
162 
163  // Fix starting position of cast to opening paren.
164  cast->pos = start;
165 
166  return std::move(cast);
167  }
168 
169  auto contents = parse_expression();
170 
171  // No closing paren.
172  if (!is_type<Tok::CloseParen>(lexer.get_tok())) {
173  throw "Expected close paren in parenthesized expression";
174  }
175 
176  // Shift closing paren.
177  lexer.shift();
178 
179  return contents;
180 }

182 AST::Expression Parser::parse_primary(void) {
183  auto tok = lexer.get_tok();
184  if (is_type<Tok::Identifier>(tok)) {
185  return parse_variable();
186  } else if (auto *i_lit = boost::get<Tok::IntLiteral>(&tok)) {
187  auto result = get_literal<AST::IntLiteral>
188  (*i_lit, lexer.get_pos());
189  lexer.shift();
190  return result;
191  } else if (auto *u_lit = boost::get<Tok::UIntLiteral>(&tok)) {
192  auto result = get_literal<AST::UIntLiteral>
193  (*u_lit, lexer.get_pos());
194  lexer.shift();
195  return result;
196  } else if (auto *f_lit = boost::get<Tok::FloatLiteral>(&tok)) {
197  auto result = get_literal<AST::FloatLiteral>
198  (*f_lit, lexer.get_pos());
199  lexer.shift();
200  return result;
201  } else if (is_type<Tok::OpenParen>(tok)) {
202  return parse_parens();
203  } else {
204  throw "Expected expression.";
205  }
206 }

211  if (auto *op = boost::get<Tok::Operator>(&tok)) {
212  auto i = precedences.find(op->op);

216  }
