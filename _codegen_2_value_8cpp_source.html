94 Value ValueGen::operator()(const AST::Binop &binop) {
95  auto lhs = visit(binop.lhs());
96  auto rhs = visit(binop.rhs());
97 
98  auto pos = binop.pos();
99 
100  if (binop.op() == "<<") {
101  return _translator.left_shift(lhs, rhs, pos);
102  } else if (binop.op() == ">>") {
103  return _translator.right_shift(lhs, rhs, pos);
104  } else if (binop.op() == "&") {
105  return _translator.bit_and(lhs, rhs, pos);
106  } else if (binop.op() == "|") {
107  return _translator.bit_or(lhs, rhs, pos);
108  } else if (binop.op() == "^") {
109  return _translator.bit_xor(lhs, rhs, pos);
110  } else if (binop.op() == "+") {
111  return _translator.add(lhs, rhs, pos);
112  } else if (binop.op() == "-") {
113  return _translator.sub(lhs, rhs, pos);
114  } else if (binop.op() == "*") {
115  return _translator.mul(lhs, rhs, pos);
116  } else if (binop.op() == "/") {
117  return _translator.div(lhs, rhs, pos);
118  } else if (binop.op() == "==") {
119  return _translator.equal(lhs, rhs, pos);
120  } else if (binop.op() == "!=") {
121  return _translator.nequal(lhs, rhs, pos);
122  } else if (binop.op() == "<") {
123  return _translator.less(lhs, rhs, pos);
124  } else if (binop.op() == "<=") {
125  return _translator.lesseq(lhs, rhs, pos);
126  } else if (binop.op() == ">") {
127  return _translator.greater(lhs, rhs, pos);
128  } else if (binop.op() == ">=") {
129  return _translator.greatereq(lhs, rhs, pos);
130  } else if (binop.op() == "&&") {
131  return _translator.bool_and(lhs, rhs, pos);
132  } else if (binop.op() == "||") {
133  return _translator.bool_or(lhs, rhs, pos);
134  } else {
135  throw Error("internal error", "unrecognized operator \"" + binop.op()
136  + "\"", pos);
137 
138  }
139 }
