69 boost::variant<double, uint64_t> Lexer::lex_number(void) {
70  uint64_t num = 0;
71 
72  /* Parse the initial series of digits. */
73  while (isdigit(c)) {
74  num *= 10;
75  num += digit(c);
76  get();
77  }
78 
79  /* If a decimal, */
80  if (c == '.') {
81  /* Parse the decimal bit. */
82  double result = (double)num;
83  double decimal_places = 0.1;
84  for (get(); isdigit(c); get()) {
85  result += decimal_places * digit(c);
86  decimal_places *= 0.1;
87  }
88 
89 
90  /* Parse the exponent if present. */
91  if (c == 'e' || c == 'E') {
92  get();
93 
94  bool neg = false;
95 
96  if (c == '-') {
97  get();
98  neg = true;
99  }
100 
101  uint64_t exp = 0;
102  for (; isdigit(c); get()) {
103  exp *= 10;
104  exp += digit(c);
105  }
106 
107  return result * pow(10.0, (neg? -1: 1) * exp);
108  }
109 
110  return result;
111  } else if (c == 'e' || c == 'E') {
112  get();
113 
114  bool neg = false;
115  if (c == '-') {
116  get();
117  neg = true;
118  }
119 
120  double result = (double)num;
121 
122  double exp = 0;
123  for (; isdigit(c); get()) {
124  exp *= 10.0;
125  exp += (double)digit(c);
126  }
127 
128  return result * pow(10.0, (neg? -1: 1) * exp);
129  } else {
130  return num;
131  }
132 }

189 void Lexer::shift(void) {
190  while (std::isspace(c)) {
191  get();
192  }
193 
194  if (c == std::char_traits<char>::eof()) {
195  eof = true;
196  return;
197  }
198 
199  /* Type name. */
200  if (isupper(c)) {
201  std::string tname;
202 
203  while (isalpha(c) || isdigit(c) || c == '_' || is_unicode(c)) {
204  tname.push_back(c);
205  get();
206  }
207 
208  tok = std::make_unique<Tok::TypeName>(tname);
209  /* Identifiers and identifier-like keywords. */
210  } else if (islower(c) || is_unicode(c)) {
211  std::string ident;
212 
213  while (isalpha(c) || isdigit(c) || c == '_' || is_unicode(c)) {
214  ident.push_back(c);
215  get();
216  }
217 
218  /* Keywords that otherwise look like identifiers. */
219  if (ident == "fn") {
220  tok = std::make_unique<Tok::Fn>();
221  } else if (ident == "struct") {
222  tok = std::make_unique<Tok::Struct>();
223  } else if (ident == "type") {
224  tok = std::make_unique<Tok::Type>();
225  } else if (ident == "return") {
226  tok = std::make_unique<Tok::Return>();
227  } else if (ident == "if") {
228  tok = std::make_unique<Tok::If>();
229  } else if (ident == "else") {
230  tok = std::make_unique<Tok::Else>();
231  } else if (ident == "while") {
232  tok = std::make_unique<Tok::While>();
233  /* If none of those, an identifier. */
234  } else {
235  tok = std::make_unique<Tok::Identifier>(ident);
236  }
237  /* Numeric literal. */
238  } else if (isdigit(c)) {
239  auto result = lex_number();
240  if (result.which() == 0) {
241  auto literal = boost::get<double>(result);
242  tok = std::make_unique<Tok::FloatLiteral>(literal);
243  } else {
244  auto literal = boost::get<uint64_t>(result);
245  tok = std::make_unique<Tok::UIntLiteral>(literal);
246  }
247  /* Operators. This is easily extensible to user-defined operators. */
248  } else if (is_opchar(c)) {
249  std::string result;
250  result.push_back(c);
251 
252  for (get(); is_opchar(c); get()) {
253  result.push_back(c);
254  }
255 
256  tok = std::make_unique<Tok::Operator>(result);
257  /* Some random syntax. */
258  } else if (c == '(') {
259  tok = std::make_unique<Tok::OpenParen>();
260  get();
261  } else if (c == ')') {
262  tok = std::make_unique<Tok::CloseParen>();
263  get();
264  } else if (c == '{') {
265  tok = std::make_unique<Tok::OpenBrace>();
266  get();
267  } else if (c == '}') {
268  tok = std::make_unique<Tok::CloseBrace>();
269  get();
270  } else if (c == ';') {
271  tok = std::make_unique<Tok::Semicolon>();
272  get();
273  } else if (c == ',') {
274  tok = std::make_unique<Tok::Comma>();
275  get();
276  } else if (c == '"') {
277  tok = std::make_unique<Tok::StringLiteral>(lex_string());
278  get();
279  } else throw Error("lexer error",
280  std::string("character \"") + c + "\" not recognized",
281  pos);
282 }
